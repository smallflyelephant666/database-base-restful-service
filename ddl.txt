-- 汇率情报
drop table if exists tbl_rate_info cascade;

create table tbl_rate_info (
  id bigint not null comment '主键'
  , publiction_id varchar(10) comment '汇率发布id'
  , currency_id varchar(20) comment '货币id'
  , tts decimal(3,2) comment 'tts'
  , ttb decimal(3,2) comment 'ttb'
  , ttm decimal(3,2) comment 'ttm'
  , del_flag tinyint not null comment '删除标志'
  , ver_no int not null comment '版本号'
  , created_by bigint not null comment '作成者id'
  , creator varchar(20) comment '作成者名'
  , updated_by bigint not null comment '更新者'
  , updater varchar(20) comment '更新者名'
  , create_time bigint not null comment '作成时间'
  , create_time_str varchar(20) comment '作成时间文字表示'
  , update_time bigint not null comment '更新时间'
  , update_time_str varchar(20) comment '更新时间文字表示'
  , constraint tbl_rate_info_pkc primary key (id)
) comment '汇率情报' ;

-- 汇率发布时间
drop table if exists tbl_rate_publiction cascade;

create table tbl_rate_publiction (
  id bigint not null comment '主键'
  , publiction_date varchar(10) comment '发布日'
  , publiction_datetime varchar(20) comment '发布日时'
  , del_flag tinyint not null comment '删除标志'
  , ver_no int not null comment '版本号'
  , created_by bigint not null comment '作成者id'
  , creator varchar(20) comment '作成者名'
  , updated_by bigint not null comment '更新者'
  , updater varchar(20) comment '更新者名'
  , create_time bigint not null comment '作成时间'
  , create_time_str varchar(20) comment '作成时间文字表示'
  , update_time bigint not null comment '更新时间'
  , update_time_str varchar(20) comment '更新时间文字表示'
  , constraint tbl_rate_publiction_pkc primary key (id)
) comment '汇率发布时间' ;

-- 币种信息
drop table if exists tbl_currency cascade;

create table tbl_currency (
  id bigint not null comment '主键'
  , code varchar(3) comment '编码'
  , name_en varchar(100) comment '货币名-英'
  , name_jp varchar(100) comment '货币名-日'
  , del_flag tinyint not null comment '删除标志'
  , ver_no int not null comment '版本号'
  , created_by bigint not null comment '作成者id'
  , creator varchar(20) comment '作成者名'
  , updated_by bigint not null comment '更新者'
  , updater varchar(20) comment '更新者名'
  , create_time bigint not null comment '作成时间'
  , create_time_str varchar(20) comment '作成时间文字表示'
  , update_time bigint not null comment '更新时间'
  , update_time_str varchar(20) comment '更新时间文字表示'
  , constraint tbl_currency_pkc primary key (id)
) comment '币种信息' ;
