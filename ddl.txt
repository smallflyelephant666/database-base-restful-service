-- 汇率发布时间
drop table if exists new_entity1 cascade;

create table new_entity1 (
  id bigint not null comment '主键'
  , publiction_id varchar(10) comment '汇率发布id'
  , currency_id varchar(20) comment '货币id'
  , tts decimal(3,2) comment 'tts'
  , ttb decimal(3,2) comment 'ttb'
  , ttm decimal(3,2) comment 'ttm'
  , del_flag tinyint not null comment '删除标志'
  , ver_no int not null comment '版本号'
  , created_by bigint not null comment '作成者id'
  , creator varchar(20) comment '作成者名'
  , updated_by bigint not null comment '更新者'
  , updater varchar(20) comment '更新者名'
  , create_time bigint not null comment '作成时间'
  , create_time_str varchar(20) comment '作成时间文字表示'
  , update_time bigint not null comment '更新时间'
  , update_time_str varchar(20) comment '更新时间文字表示'
  , constraint new_entity1_pkc primary key (id)
) comment '汇率发布时间' ;
